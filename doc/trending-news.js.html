(function() {
  var TrendingNews, config, hashCode, http, logger, storage;

  http = require('http');

  storage = require('node-persist');

  logger = require('../lib/logger');

  hashCode = require('../lib/hash-code');

  config = require('../lib/config');

  require('../lib/mock-index-stream');


  /**
    * @classdesc Gets up-to-date, trending news
    *
    * @param [runMode='default'] {string} determines logging level and whether to use Nock
    * @param [scoreThreshold=100] {number} lowest allowed trending score for any news item
    *
    * @example How to create an instance, method(1)
    *   new TrendingNews(runMode, scoreThreshold)
    *
    * @example How to create an instance, method(2)
    *   new TrendingNews(runMode)
    *
    * @example How to create an instance, method(3)
    *   new TrendingNews(scoreThreshold)
    *
    * @property {object} results - latest news results [readonly]
    * @property {boolean} finished - if class instance is done processing [readonly]
    *
    * @class TrendingNews
   */

  TrendingNews = (function() {
    var filterNewsByTrendScore, filterNewsIfSeenBefore, finished, get, getLatestNewsForTopic, results, resultsCallback, scoreThreshold, set;

    get = function(props) {
      var getter, name, _results;
      _results = [];
      for (name in props) {
        getter = props[name];
        _results.push(TrendingNews.prototype.__defineGetter__(name, getter));
      }
      return _results;
    };

    set = function(props) {
      var name, setter, _results;
      _results = [];
      for (name in props) {
        setter = props[name];
        _results.push(TrendingNews.prototype.__defineSetter__(name, setter));
      }
      return _results;
    };

    scoreThreshold = 100;

    results = null;

    get({
      results: function() {
        return results;
      }
    });

    finished = false;

    get({
      finished: function() {
        return finished;
      }
    });


    /**
      * @description Constructs a new framework for getting news
      * @constructs TrendingNews
     */

    function TrendingNews(mode, score) {
      if (mode == null) {
        mode = 'default';
      }
      if (score == null) {
        score = 100;
      }
      if (!mode) {
        mode = 'default';
      }
      if (!score) {
        score = 100;
      }
      if (mode === 'debug') {
        logger.debugLevel = 'info';
      } else if (mode === 'test') {
        logger.debugLevel = 'error';
      } else if (mode === 'prod') {
        logger.debugLevel = 'info';
      } else {
        mode = 'default';
      }
      if (mode === 'prod') {
        process.env.NOCK_OFF = true;
        logger.log('warn', 'Nock is OFF!');
      } else {
        logger.log('warn', 'Nock is ON');
      }
      logger.log('warn', 'in ' + mode + 'Mode (lowest log level: ' + logger.debugLevel + ')');
      scoreThreshold = score;
      logger.log('warn', 'scoreThreshold = ' + scoreThreshold);
    }


    /**
      * @description Removes news items that fall below the trending score threshold
      *
      * @param newsItems {Array<Object>} list of news items to filter
      * @return {Array<Object>} list of news items, filtered by trending score
      *
      * @method filterNewsByTrendScore
      * @memberof TrendingNews
      * @instance
      * @private
     */

    filterNewsByTrendScore = function(newsItems) {
      return newsItems.filter(function(item) {
        return item.trending_score >= scoreThreshold;
      });
    };


    /**
      * @description Removes news items that have been seen before, determined by a hash of its title
      *
      * @param newsItems {Array<Object>} list of news items to filter
      * @return {Array<Object>} list of news items, filtered by seen before status
      *
      * @method filterNewsIfSeenBefore
      * @memberof TrendingNews
      * @instance
      * @private
     */

    filterNewsIfSeenBefore = function(newsItems) {
      var TypeError, item, seenWhen, titleHash, unseen, _i, _len;
      unseen = [];
      storage.initSync({
        dir: config.STORAGE_DIR
      });
      for (_i = 0, _len = newsItems.length; _i < _len; _i++) {
        item = newsItems[_i];
        titleHash = hashCode.hash(item.title);
        logger.log('info', titleHash + ', ' + item.title);
        seenWhen = storage.getItem(titleHash);
        if (seenWhen === void 0) {
          unseen.push(item);
          seenWhen = [new Date(Date.now())];
        } else {
          try {
            seenWhen.push(new Date(Date.now()));
          } catch (_error) {
            TypeError = _error;
            logger.log('warn', 'Problem persisting storage for item ' + titleHash + ' in a previous cycle - marking as seen');
            seenWhen = [new Date(Date.now())];
          }
        }
        storage.setItem(titleHash, seenWhen);
      }
      return unseen;
    };


    /**
      * @description Makes get call to an API for news about a topic
      *
      * @param topic {String} topic to get news about
      * @param successCallback {Function} function to call on the success of request
      *
      * @method getLatestNewsForTopic
      * @memberof TrendingNews
      * @instance
      * @private
     */

    getLatestNewsForTopic = function(topic, successCallback) {
      var classObj, options;
      logger.log('info', 'Getting all news for topic: ' + topic + '...');
      classObj = this;
      options = {
        hostname: config.REQUEST_HOSTNAME,
        path: config.REQUEST_PATH + encodeURIComponent(topic),
        headers: config.REQUEST_HEADERS
      };
      return http.get(options, function(response) {
        var data;
        data = '';
        response.on('data', function(chunk) {
          return data += chunk;
        });
        response.on('end', function() {
          var SyntaxError, allNewsItems, filteredItems, unseenItems;
          if (response.statusCode === 200) {
            try {
              data = JSON.parse(data);
            } catch (_error) {
              SyntaxError = _error;
              logger.log('error', "Response '" + data + "' is not valid JSON!");
              data = {
                link_list: []
              };
            }
            allNewsItems = data.link_list;
            logger.log('info', '# ' + topic + ' items before filter: ' + allNewsItems.length);
            filteredItems = filterNewsByTrendScore(allNewsItems);
            logger.log('info', '# ' + topic + ' items after filter: ' + filteredItems.length);
            unseenItems = filterNewsIfSeenBefore(filteredItems);
            logger.log('info', '# ' + topic + ' items never seen before: ' + unseenItems.length);
            logger.log('info', '---');
            return successCallback.call(classObj, topic, unseenItems);
          } else {
            return classObj.handleBadResponse(topic, response.statusCode);
          }
        });
        return response.on('error', function(e) {
          return classObj.handleError(topic, e.message, 'response');
        });
      }).on('error', function(e) {
        return classObj.handleError(topic, e.message, 'request');
      });
    };


    /**
      * @description Stores the resulting news items of a topic.
      * Waits for the results for all topics before handling
      *
      * @param topic {String} topic to get news about
      * @param result {Array<Object>} 
      *
      * @callback TrendingNews~resultsCallback
      * @memberof TrendingNews
      * @instance
      * @private
     */

    resultsCallback = function(topic, result) {
      var classObj;
      classObj = this;
      results[topic] = result;
      if (Object.keys(results).length === config.TOPICS.length) {
        classObj.logResults(results);
        return finished = true;
      }
    };


    /**
      * @description Logs results of processing
      * Logs with level 'warn', public to allow for unit testing, although marked 'private'
      *
      * @param res {Object} results to log
      *
      * @method logResults
      * @memberof TrendingNews
      * @instance
      * @private
     */

    TrendingNews.prototype.logResults = function(res) {
      return logger.log('warn', res);
    };


    /**
      * @description Logs errors that occur due to a 'bad' response status code.
      * Public to allow for unit testing, although marked 'private'
      *
      * @param topic {String} topic with bad response status code
      * @param statusCode {Number} bad response status code
      *
      * @method handleBadResponse
      * @memberof TrendingNews
      * @instance
      * @private
     */

    TrendingNews.prototype.handleBadResponse = function(topic, statusCode) {
      return logger.log('error', 'Response with status code ' + statusCode + ' for topic ' + topic);
    };


    /**
      * @description Logs errors that occur due to a bad request or response.
      * Public to allow for unit testing, although marked 'private'
      *
      * @param topic {String} topic being processed during error
      * @param message {String} error message
      * @param httpObjType {String} request or response, depending on when error occurred
      *
      * @method handleError
      * @memberof TrendingNews
      * @instance
      * @private
     */

    TrendingNews.prototype.handleError = function(topic, message, httpObjType) {
      return logger.log('error', 'Problem with ' + httpObjType + ' for topic ' + topic + '... ' + message);
    };


    /**
      * @description Processes every news topic in an asynchronous manner
      *
      * @method getLatest
      * @memberof TrendingNews
      * @instance
     */

    TrendingNews.prototype.getLatest = function() {
      var topic, _i, _len, _ref, _results;
      results = {};
      logger.log('info', 'Getting latest trending news items for ' + config.TOPICS.length + ' topic(s)...');
      _ref = config.TOPICS;
      _results = [];
      for (_i = 0, _len = _ref.length; _i < _len; _i++) {
        topic = _ref[_i];
        _results.push(getLatestNewsForTopic.call(this, topic, resultsCallback));
      }
      return _results;
    };

    return TrendingNews;

  })();
